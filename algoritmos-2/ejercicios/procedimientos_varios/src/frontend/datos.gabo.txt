Procedimiento datos(E/S: v[t], i, t: Entero);
Inicio
  var n: Entero;
  Mostrar << "Ingrese un tamano para el vector (2 - 500) ";
  Leer >> t;

  Mientras (t < 2 Or t > 500) Hacer;
    Mostrar << t, " esta fuera del rango de 2 a 500. Ingrese otro numero ";
    Leer >> t;
  Fin_Mientras

  Para (i = 1 Hasta i = t) Hacer;
    Mostrar << "Ingrese un numero entero positivo ";
    Leer >> n;

    Mientras (n < 1) Hacer;
      Mostrar << n, " no es un numero positivo. Ingrese otro numero ";
      Leer >> n;
    Fin_Mientras

    v[i] = n;
  Fin_Para
Fin_Procedimiento
